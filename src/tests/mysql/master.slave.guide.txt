1/ docker pull mysql:8.0
2/ docker network create mysql-master-slave-demo
3/ docker run -d --name mysql-master -p 8811:3306 -e MYSQL_ROOT_PASSWORD=Password1! --network mysql-master-slave-demo mysql:8.0
4/ docker run -d --name mysql-slave -p 8822:3306 -e MYSQL_ROOT_PASSWORD=Password1! --network mysql-master-slave-demo mysql:8.0
5/ update config for mysql-master:/etc/my.cnf
    log_bin=mysql-bin
    server-id=1
6/ update config for mysql-slave:/etc/my.cnf
    log_bin=mysql-bin
    server-id=2
7/ docker exec -it mysql-master bash
    - sign in: 
        mysql -uroot -p
        Password1!
        SHOW MASTER STATUS
    
        -> get master_log_file and master_log_pos information

8/ docker inspect mysql-master ->  get master ip address

9/ docker exec -it mysql-slave bash
    - sign in: 
        mysql -uroot -p
        Password1!
       
        CHANGE MASTER TO
        MASTER_HOST = '172.20.0.3',
        MASTER_PORT = 3306,
        MASTER_USER = 'root',
        MASTER_PASSWORD = 'Password1!',
        master_log_file = 'mysql-bin.000001',
        master_log_pos = 157,
        master_connect_retry = 60,
        GET_MASTER_PUBLIC_KEY = 1;
